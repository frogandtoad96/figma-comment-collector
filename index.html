  <p>Figma Token과 File Key를 입력하면 댓글을 수집해 노션용/피그마용 문서를 생성합니다.</p>


  <label for="token">Figma Token</label>

  <input id="token" type="password" placeholder="figd_..." />

  <input id="fileKey" type="text" placeholder="aPZLhcBaXZbOfzxjR9FXq5" />


  <div class="row">

    <button id="runBtn">실행</button>

  </div>


  <div class="result hidden" id="resultBox">

    <h2>change-log.md</h2>

    <pre id="changeLog"></pre>


    <h2>figma-todo.md</h2>

    <pre id="figmaTodo"></pre>


    <div class="row">

      <button id="downloadChangeLog">change-log.md 다운로드</button>

      <button id="downloadFigmaTodo">figma-todo.md 다운로드</button>

    </div>

  </div>


  <script>

    const runBtn = document.getElementById("runBtn");

    const resultBox = document.getElementById("resultBox");

    const changeLogEl = document.getElementById("changeLog");

    const figmaTodoEl = document.getElementById("figmaTodo");

    const downloadChangeLogBtn = document.getElementById("downloadChangeLog");

    const downloadFigmaTodoBtn = document.getElementById("downloadFigmaTodo");


    let latestChangeLog = "";

    let latestFigmaTodo = "";


    function downloadText(filename, text) {

      const blob = new Blob([text], { type: "text/markdown;charset=utf-8" });

      const url = URL.createObjectURL(blob);

      const a = document.createElement("a");

      a.href = url;

      a.download = filename;

      a.click();

      URL.revokeObjectURL(url);

    }


    downloadChangeLogBtn.addEventListener("click", () => {

      downloadText("change-log.md", latestChangeLog);

    });


    downloadFigmaTodoBtn.addEventListener("click", () => {

      downloadText("figma-todo.md", latestFigmaTodo);

    });


    runBtn.addEventListener("click", async () => {

      const token = document.getElementById("token").value.trim();

      const fileKey = document.getElementById("fileKey").value.trim();


      if (!token || !fileKey) {

        alert("Token과 File Key를 모두 입력해주세요.");

        return;

      }


      runBtn.disabled = true;

      runBtn.textContent = "실행 중...";


      try {

        const res = await fetch("/api/report", {

          method: "POST",

          headers: { "Content-Type": "application/json" },

          body: JSON.stringify({ token, fileKey })

        });


        const data = await res.json();


        if (!res.ok) {

          throw new Error(data.error || "실행 중 오류가 발생했습니다.");

        }


        latestChangeLog = data.changeLog;

        latestFigmaTodo = data.figmaTodo;


        changeLogEl.textContent = latestChangeLog;

        figmaTodoEl.textContent = latestFigmaTodo;

        resultBox.classList.remove("hidden");

      } catch (err) {

        alert(err.message);

      } finally {

        runBtn.disabled = false;

        runBtn.textContent = "실행";

      }

    });

  </script>
